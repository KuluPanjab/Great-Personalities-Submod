# Sets Andrew Jackson as President in 1830
country_event = {
	
	id = 1010099
	
	title = "Dynamic Leaders: Setting Andrew Jackson as president" 
	desc = "Dynamic Leaders: Setting Andrew Jackson as president"
	picture = "danishgovernment"
	
	fire_only_once = yes 
	
	trigger = { 
		
		tag = JAN
		
		AND = {
			year = 1836
			NOT = { year = 1837 }
		}
		
		USA = {
			
			OR = {
				government = democracy
				has_country_flag = democracy_gov
			}
			
			NOT = { has_country_modifier = andrew_jackson }
			
		}
		
	}
	
	mean_time_to_happen = { days = -1 }
	
	option = {
		
		name = "Setting..."
		
		USA = {
			
			add_country_modifier = { 
				name = andrew_jackson
				duration = -1
			}
			
		}
		
	}
	
}

# Native American Cultures Reduction
country_event = {
	
	id = 1010100
	
	title = "Dynamic Leaders: Reducing Native American Cultures" 
	desc = "Dynamic Leaders: Reducing Native American Cultures" 
	
	trigger = { 
		
		tag = JAN
		
		USA = { has_country_modifier = andrew_jackson }
		
	}
	
	mean_time_to_happen = { months = 1 }
	
	option = {
		
		name = "Reducing..."
		
		USA = {
			
			any_pop = {
				
				limit = { is_culture_group = native_american }
				
				reduce_pop = 0.01
				
			}
			
		}
		
	}
	
}

# 1836 Election hold on Historical Date
country_event = {
	
	id = 1010101
	
	title = "Dynamic Leaders: Holding the 1836 USA election" 
	desc = "Dynamic Leaders: Holding the 1836 USA election"
	picture = "danishgovernment"
	
	fire_only_once = yes 
	
	trigger = { 
		
		tag = JAN
		
		AND = {
			year = 1836
			month = 10
		}
		
		USA = {
			
			OR = {
				government = democracy
				has_country_flag = democracy_gov
			}
			
		}
		
	}
	
	mean_time_to_happen = { days = 2 }
	
	option = {		
		name = "Holding..."
		
		USA = {
			
			election = yes
			
			set_country_flag = 1836_presidential_election
			
		}
		
	}
	
}
