# Sets Andrew Jackson as President in 1830
country_event = {
	
	id = 1010099
	
	title = "Dynamic Leaders: Setting Andrew Jackson as president" 
	desc = "Dynamic Leaders: Setting Andrew Jackson as president"
	picture = "danishgovernment"

	fire_only_once = yes
	
	trigger = { 
		
		tag = JAN
		
		AND = {
			year = 1836
			NOT = { year = 1837 }
		}
		
		USA = {
			
			OR = {
				government = democracy
				has_country_flag = democracy_gov
			}
			
			NOT = { has_country_modifier = andrew_jackson }
			
		}
		
	}
	
	mean_time_to_happen = { days = -1 }
	
	option = {
		
		name = "Setting..."
		
		USA = {
			
			add_country_modifier = { 
				name = andrew_jackson
				duration = -1
			}
			
		}
		
	}
	
}
