# Native American Cultures Reduction
country_event = {
	id = 1010100
	title = "Dynamic Leaders: Reducing Native American Cultures" 
	desc = "Dynamic Leaders: Reducing Native American Cultures" 
	trigger = { 
		tag = JAN
		USA = { has_country_modifier = andrew_jackson }
	}
	mean_time_to_happen = { months = 1 }
	option = {
		name = "Reducing..."
		USA = {
			any_pop = {
				limit = { is_culture_group = native_american }
				reduce_pop = 0.95
			}
		}
	}
}
